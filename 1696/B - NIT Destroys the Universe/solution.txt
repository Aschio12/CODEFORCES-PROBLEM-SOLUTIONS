for _ in range(int(input())):    n=int(input())    arr=list(map(int,input().split()))    i=0    while i<n:       if arr[i]!=0:           break       i+=1    if i==n:        print(0)    else:        start,end=-1,-1        for j in range(n):            if arr[j]!=0:                start=j                break        for k in range(n-1,-1,-1):             if arr[k]!=0:                 end=k                 break        found=False        for i in range(start,end+1):            if arr[i]==0:                found=True                 break        if found:            print(2)        else:            print(1)
