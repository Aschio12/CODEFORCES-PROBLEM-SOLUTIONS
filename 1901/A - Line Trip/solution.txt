for _ in range(int(input())):    n,x=map(int,input().split())    station=list(map(int,input().split()))    ans=station[0]    if n>=2:        for i in range(1,n):            ans=max(ans,station[i]-station[i-1])        ans=max(ans,(x-station[n-1])*2)    else:        ans=max(station[0],(x-station[0])*2)       print(ans)
